const dialog = document.querySelector('div[role="dialog"]');

if (!dialog) {
  console.error("âŒ Dialog not found");
} else {
  console.log("âœ… Dialog found");

  const scrollable = [...dialog.querySelectorAll('div')].find(div => {
    const style = getComputedStyle(div);
    return (style.overflowY === 'scroll' || style.overflowY === 'auto') && div.scrollHeight > div.clientHeight;
  });

  if (!scrollable) {
    console.error("âŒ No scrollable area found");
  } else {
    console.log("âœ… Scrollable area found:", scrollable);

    let lastScrollTop = -1;
    let sameScrollCount = 0;
    const maxSameScrolls = 3;

    const scrollInterval = setInterval(() => {
      scrollable.scrollTop = scrollable.scrollHeight;

      // Simulate scroll event (triggers lazy-loaders)
      scrollable.dispatchEvent(new Event('scroll', { bubbles: true }));

      console.log("ðŸ“œ Scrolling... scrollTop:", scrollable.scrollTop);

      if (scrollable.scrollTop === lastScrollTop) {
        sameScrollCount++;
      } else {
        sameScrollCount = 0;
      }

      lastScrollTop = scrollable.scrollTop;

      if (sameScrollCount >= maxSameScrolls) {
        clearInterval(scrollInterval);
        console.log("âœ… Done scrolling, all followers likely loaded.");
      }
    }, 2000);
  }
}
